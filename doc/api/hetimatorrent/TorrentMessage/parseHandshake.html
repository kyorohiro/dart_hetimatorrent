static Future<TorrentMessage> parseHandshake(EasyParser parser, [int maxOfMessageSize = 256 * 1024]) async {
  parser.push();
  try {
    MessageHandshake message = await MessageHandshake.decode(parser);
    parser.pop();
    return message;
  } catch (e) {
    parser.back();
    parser.pop();
    throw e;
  }
}
