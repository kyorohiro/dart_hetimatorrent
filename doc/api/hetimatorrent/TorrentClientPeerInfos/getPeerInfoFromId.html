TorrentClientPeerInfo getPeerInfoFromId(int id) {
  List<TorrentClientPeerInfo> targetPeers = _peerInfos.getWithFilter((TorrentClientPeerInfo info) {
    return (info.id == id);
  });
  return (targetPeers.length > 0 ? targetPeers.first : null);
}
