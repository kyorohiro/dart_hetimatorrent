static List<int> toCompactNodeInfos(List<KPeerInfo> infos) {
  List<int> ret = [];
  for (KPeerInfo info in infos) {
    ret.addAll(info.toCompactNodeInfo());
  }
  return new Uint8List.fromList(ret);
}
