async.Future<StartResult> start() async {
  if (outputLog) {
    print("TrackerServer#start");
  }
  _server.basePort = port;
  _server.numOfRetry = 0;
  if (res == null) {
    res = _server.onResponse.listen(onListen);
  }
  await _server.startServer();
  _isStart = true;
  return new StartResult();
}
