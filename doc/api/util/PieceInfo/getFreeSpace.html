List<int> getFreeSpace(int size) {
  int begin = 0;
  for(PieceInfoItem info in mInfo) {
    if(info.start > begin && begin != info.start) {
      return [begin, info.start];
    } else {
      begin = info.end;
    }
  }
  return [begin, begin+size];
}
