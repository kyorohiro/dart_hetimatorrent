List<TorrentClientPeerInfo> getPeerInfoFromXx(Function filter) {
  List<TorrentClientPeerInfo> ret = [];
  for (TorrentClientPeerInfo info in this.peerInfos) {
    if (true == filter(info)) {
      ret.add(info);
    }
  }
  return ret;
}
